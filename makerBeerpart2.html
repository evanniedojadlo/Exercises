/** * Exercise 2 Part 2: getAge **/ function calculateAge(birthMonth, birthDay, birthYear) { // Get Todays Date var todayDate = new Date(), todayYear = todayDate.getFullYear(), todayMonth = todayDate.getMonth(), todayDay = todayDate.getDate(); // Calculate Age var age = todayYear - birthYear; // Check month discrepancies if (todayMonth < (birthMonth - 1)) { age--; } // Check day discrepancies if (((birthMonth - 1) == todayMonth) && (todayDay < birthDay)) { age--; } // @return {Number} age return age; }
